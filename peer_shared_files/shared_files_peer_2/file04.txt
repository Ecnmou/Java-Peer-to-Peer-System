#full, null, without intercept models
fullmodel = lm(Income~., testData)
nullmodel = lm(Income~1, testData)
model = update(fullmodel,.~.-1)

summary(fullmodel)
summary(model) #I'll work with this, R=0.77>0.2


#-----Conditions
#normality -> NO
stdmodel = rstandard(model)
shapiro.test(stdmodel) 
lillie.test(stdmodel) 

#homoskedasticity -> NO
quantcut = function(x, digits=6) { cut(x, breaks=quantile(x), include.lowest=TRUE, dig.lab=digits) }
qfits = quantcut(model$fit)
leveneTest(stdmodel, qfits) 

#NOT correlation -> YES
dwt(model)
durbinWatsonTest(model)


#-----Transformation needed, firstly I aim at fixing heteroscedasticity because is more important and it affects normality too.
#-----Out of all the tranformations applied(y'=1/y or ln(y) or arcsin(sqrt(y)) or...), the best was y' = sqrt(y). 
#-----WLS could be used too, but it solely fixes heteroscedasticity and not normality
#model with all 3 conditions
model = lm(sqrt(Income) ~ Gender + Vote + Political_orientation + Age + Marital_Status + Years_of_residence - 1, testData)
stdmodel = rstandard(model)

#normality -> YES
shapiro.test(stdmodel) 
lillie.test(stdmodel)
plot(model, which=2)

#homoskedasticity -> YES
qfits = quantcut(model$fit)
leveneTest(stdmodel, qfits)
plot(model, which=1, pch=16, lwd=1.5)
plot(model, which=3, pch=16, lwd=1.5)
plot(model$fit, stdmodel, ylim = c(-4, 4), pch=16)

#NOT correlation -> YES

model


#-----Variable selection, all possible combinations of stepwise procedures
n = dim(data)[1]
#stepwise
modelstep = step(model, direction = "both", k = 2) #AIC
modelstep2 = step(model, direction = "both", k = log(n)) #BIC
modelstep3 = step(nullmodel, scope=list(lower=nullmodel, upper=model), direction = "both", k = 2) #AIC
modelstep4 = step(nullmodel, scope=list(lower=nullmodel, upper=model), direction = "both", k = log(n)) #BIC

#backward
modelstep5 = step(model, direction = "back", k = 2) #AIC
modelstep6 = step(model, direction = "back", k = log(n)) #BIC

#forward
modelstep7 = step(nullmodel, scope=list(lower=nullmodel, upper=model), direction = "forward", k = 2) #AIC
modelstep8 = step(nullmodel, scope=list(lower=nullmodel, upper=model), direction = "forward", k = log(n)) #BIC


#-----Multicollinearity, updating existing possible models
vif(model)#

modelstep2 = update(modelstep2,.~.-Gender)

modelstep5 = update(modelstep5,.~.-Gender)
vif(modelstep6)#
modelstep6 = update(modelstep6,.~.-Gender)
vif(modelstep7)
vif(modelstep8)


#-----Choosing the optimal model, still model which might have a multicollinearity problem, yet is the
