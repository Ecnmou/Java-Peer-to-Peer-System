#(i)
#Experiment's data
nonP = c(22.1, 22.3, 26.2, 29.6, 31.7, 33.5, 38.9, 39.7, 43.2, 43.2)
beg = c(32.5, 37.1, 39.1, 40.5, 45.5, 51.3, 52.6, 55.7, 55.9, 57.7)
adv = c(40.1, 45.6, 51.2, 56.4, 58.1, 71.1, 74.9, 75.9, 80.3, 85.3)

score = c(nonP, beg, adv)
group = as.factor(rep(1:3, each=length(adv)))
levels(group) = c("Non Player", "Beginner", "Advanced")
data = data.frame(score = score, group = group)

#Anova test
anova <- aov(score ~ group, data)
summary(anova)

#---checking conditions
#...Normality of the residuals
lillie.test(anova$res)
shapiro.test(anova$res)
#Both p-values>0.1 so we cannot reject normality hypothesis.

qqnorm(anova$res)
qqline(anova$res)

#...Homogeneity of Variance
leveneTest(anova)
#p-value<0.01 => homoscedasticity hypothesis is rejected for a=0.01
bartlett.test(score ~ group, data)
#p-value<0.1 => homoscedasticity hypothesis is rejected for a=0.1

#requested test
oneway.test(score~group, data=data, var.equal = T)
oneway.test(score~group, data=data, var.equal = F)
#Variances can be equal or not depending on a(level of significance), but in both cases
#H0 is rejected(really small p-value) => there is statistically significant difference in
#the ability to remember between the groups.


#(ii)
#Tukey test
TukeyHSD(anova)
plot(TukeyHSD(anova))
#There is a significant difference between tournament players and non-players

#(iii)
pairwise.t.test(data$score ,data$group)

#--With bonferroni correction--#
pairwise.t.test(data$score, data$group, p.adjust.method = "bonferroni")

#pairwise.t.test consists of 3 subtests, Non player-Beginner, 
#Beginner-Advanced and Non player-Advanced.
#In both t.tests we can reject H0(showing this way that there is
#statistically significant difference in the ability to remember between the groups),
#in the Non player-Beginner test for a=0.05, in the Non player-Advanced always and
#in the Beginner-Advanced for a=0.01. The only thing that changes is the p-value
#adjustment method(holm or bonferroni).

#--Not equal variances--#
pairwise.t.test(data$score, data$group, pool.sd = F, p.adjust.method = "bonferroni")
#Same test as above, in case our variances are different.
