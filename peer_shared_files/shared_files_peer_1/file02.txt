#------------------ Categorical-Quantitative Variables ------------------#

#-----Gender~Age, Years of Residence, Income-----#
#2 subcategories -> t.tests, wilcoxon tests

tapply(data$Age, data$Gender, shapiro.test) #NOT normal -> wilcoxon
leveneTest(data$Age, data$Gender) #homoscedasticity OK
tapply(data$Age, data$Gender, skew) #neglibible skew -> I'll try t.test too
t.test(Age~Gender, data, var.equal=T) #NO significant difference between the sexes
plotmeans(Age~Gender, data, connect=F, main="Age~Gender ErrorBars",)
wilcox.test(Age~Gender, data) #NO significant difference between the sexes
boxplot(Age~Gender, data, main="Age~Gender BoxPlot", col=2:3) 


tapply(data$Years_of_residence, data$Gender, shapiro.test) #NOT normal -> wilcoxon
leveneTest(data$Years_of_residence, data$Gender) #homoscedasticity OK
tapply(data$Years_of_residence, data$Gender, skew) #mean is not the proper measure
wilcox.test(Years_of_residence~Gender, data) #NO significant difference between the sexes
boxplot(Years_of_residence~Gender, data, main="Years of Residence~Gender BoxPlot", col=2:3)


tapply(data$Income, data$Gender, shapiro.test) #NOT normal -> wilcoxon
leveneTest(data$Income, data$Gender) #homoscedasticity OK for a=0.05
tapply(data$Income, data$Gender, skew) #mean is not the proper measure
wilcox.test(Income~Gender, data) #YES significant difference between the sexes
boxplot(Income~Gender, data, main="Income~Gender BoxPlot", col=2:3) 




#-----Vote~Age, Years of Residence, Income-----#
#>2 subcategories -> anova, Kruskal-Wallis tests

#Anova
anova = aov(Age~Vote, data)
summary(anova)
#NOT normal -> Kruskal-Wallis(categories with <50 observations)
shapiro.test(anova$residuals)
lillie.test(anova$residuals)
qqnorm(anova$residuals)
qqline(anova$residuals, col=2)
#Homoscedasticity OK 
leveneTest(anova)
#YES significant difference between the voters
kruskal.test(Age~Vote, data) 
boxplot(Age~Vote, data, main="Vote~Age BoxPlot", col=2:7)
